 75:     {
 76:         $delegateBuilder = new MockDelegateFunctionBuilder();

 78:         
 79:         $mock = $this->getMockBuilder($delegateBuilder->getFullyQualifiedClassName())->getMockForAbstractClass();
 80:         
 81:         $mock->__phpunit_getInvocationMocker()->addMatcher(new DefaultArgumentRemover());
 82:         
 83:         $functionMockBuilder = new MockBuilder();
 84:         $functionMockBuilder->setNamespace($namespace)
 85:                             ->setName($name)
 86:                             ->setFunctionProvider($mock);
 87:                 
 88:         $functionMock = $functionMockBuilder->build();
 89:         $functionMock->enable();
 90:         
 91:         $this->registerForTearDown($functionMock);
 92:         
 93:         $proxy = new MockObjectProxy($mock);

 95:     }
